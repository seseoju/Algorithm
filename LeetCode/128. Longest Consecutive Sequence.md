```py
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        nums_set = set(nums)
        result = 0
        
        for num in nums_set:
            if num - 1 in nums_set:
                continue
            
            cnt = 1
            curr = num
            while True:
                if curr + 1 in nums_set:
                    cnt += 1
                    curr += 1
                else:
                    break
            result = max(result, cnt)

        return result
```
